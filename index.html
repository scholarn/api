    // Show message (success or error) function showMessage(message, type = "error") { const box = document.getElementById("message-box"); box.innerText = message; box.className = `mb-4 w-full max-w-md p-4 rounded text-white ${type === 'error' ? 'bg-red-500' : 'bg-green-500'}`; box.classList.remove("hidden"); setTimeout(() => box.classList.add("hidden"), 4000); } // Fetch all books async function fetchBooks() { try { const res = await fetch(API_URL); if (!res.ok) throw new Error("Failed to fetch books."); const books = await res.json(); renderBooks(books); } catch (err) { showMessage(err.message); } } // Render books to the list function renderBooks(books) { const list = document.getElementById("books-list"); list.innerHTML = ""; books.forEach(book => { const li = document.createElement("li"); li.className = "bg-white p-4 rounded shadow flex justify-between items-center"; li.innerHTML = `

    ${book.title}

    ${book.author}

    `; list.appendChild(li); }); } // Add or update book document.getElementById("book-form").addEventListener("submit", async (e) => { e.preventDefault(); const title = document.getElementById("title").value; const author = document.getElementById("author").value; try { if(editBookId) { const res = await fetch(`${API_URL}/${editBookId}`, { method: "PUT", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ title, author }) }); if (!res.ok) throw new Error("Failed to update book."); showMessage("Book updated successfully!", "success"); editBookId = null; document.getElementById("form-title").innerText = "Add Book"; document.getElementById("cancel-edit").classList.add("hidden"); } else { const res = await fetch(API_URL, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ title, author }) }); if (!res.ok) { const errData = await res.json(); throw new Error(errData.errors ? JSON.stringify(errData.errors) : "Failed to add book."); } showMessage("Book added successfully!", "success"); } document.getElementById("book-form").reset(); fetchBooks(); } catch (err) { showMessage(err.message); } }); // Edit a book function editBook(id, title, author) { editBookId = id; document.getElementById("title").value = title; document.getElementById("author").value = author; document.getElementById("form-title").innerText = "Edit Book"; document.getElementById("cancel-edit").classList.remove("hidden"); } // Cancel edit document.getElementById("cancel-edit").addEventListener("click", () => { editBookId = null; document.getElementById("book-form").reset(); document.getElementById("form-title").innerText = "Add Book"; document.getElementById("cancel-edit").classList.add("hidden"); }); // Delete a book async function deleteBook(id) { if(!confirm("Are you sure you want to delete this book?")) return; try { const res = await fetch(`${API_URL}/${id}`, { method: "DELETE" }); if (!res.ok) throw new Error("Failed to delete book."); showMessage("Book deleted successfully!", "success"); fetchBooks(); } catch (err) { showMessage(err.message); } } // Initial load fetchBooks();
